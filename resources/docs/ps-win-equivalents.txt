    Set-Alias -Name which -Value Get-Command
    
    Set-Alias aliasName targetName

    Alias

      Get-Alias theAliasName
      Get-Alias -Name theAliasName

[wc]
  wc -l
    Get-Content file | Measure-Object -Line

  wc -w
    Get-Content file | Measure-Object -Word

  wc -m
    Get-Content file | Measure-Object -Character

  wc -L file
    Get-Content file | Measure-Object -Maximum -Property Length
